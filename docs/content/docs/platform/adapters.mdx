## How adapters work

The `kavachos` package has zero framework dependencies. It works with the Web platform `Request`/`Response` API. Adapter packages wrap the core and expose framework-idiomatic handlers for authentication, authorization, and MCP OAuth routes.

Each adapter accepts a `kavach` instance and an optional config object:

```typescript
kavachHono(kavach, {
  mcp?: McpModule,     // pass to enable MCP OAuth endpoints
  basePath?: string,   // defaults to '/api/kavach'
})
```

The adapter registers all REST routes for agents, permissions, audit, and delegation under `basePath`. Pass a `createMcpModule(kavach)` instance to also mount the MCP OAuth endpoints.

## Framework setup

### Installation

Install only the adapter you need:

```bash
npm install @kavachos/hono        # Hono
npm install @kavachos/express     # Express
npm install @kavachos/nextjs      # Next.js App Router
npm install @kavachos/fastify     # Fastify
npm install @kavachos/nuxt        # Nuxt
npm install @kavachos/sveltekit   # SvelteKit
npm install @kavachos/astro       # Astro
```

### Usage

<Tabs items={['Hono', 'Express', 'Next.js', 'Fastify', 'Nuxt', 'SvelteKit', 'Astro']}>
  <Tab value="Hono">
    `kavachHono` returns a Hono router. Mount it with `app.route`.

    ```typescript
    import { Hono } from 'hono';
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachHono } from '@kavachos/hono';

    const kavach = createKavach({
      database: { provider: 'sqlite', url: 'kavach.db' },
      baseUrl: 'https://auth.yourapp.com',
      mcp: {
        issuer: 'https://auth.yourapp.com',
        audience: 'https://mcp.yourapp.com',
      },
    });

    const mcp = createMcpModule(kavach);

    const app = new Hono();

    // Mount all KavachOS routes (agents, permissions, audit, MCP OAuth)
    app.route('/api/kavach', kavachHono(kavach, { mcp }));

    // Protect MCP tool routes
    const { withMcpAuth } = kavachHono(kavach, { mcp });
    app.use('/mcp/*', withMcpAuth());

    app.get('/mcp/tools/list', (c) => {
      const agent = c.get('agent');   // AgentIdentity set by middleware
      return c.json({ tools: [] });
    });

    export default app;
    ```
  </Tab>

  <Tab value="Express">
    `kavachExpress` returns an Express Router. Mount it with `app.use`.

    ```typescript
    import express from 'express';
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachExpress } from '@kavachos/express';
    import type { AgentIdentity } from 'kavachos';

    const kavach = createKavach({
      database: { provider: 'postgres', url: process.env.DATABASE_URL! },
      baseUrl: process.env.AUTH_BASE_URL!,
      mcp: {
        issuer: process.env.AUTH_BASE_URL!,
        audience: process.env.MCP_BASE_URL!,
      },
    });

    const mcp = createMcpModule(kavach);

    const app = express();
    app.use(express.json());

    // Mount all KavachOS routes
    app.use('/api/kavach', kavachExpress(kavach, { mcp }));

    // Protect MCP tool routes
    const { withMcpAuth } = kavachExpress(kavach, { mcp });
    app.use('/mcp', withMcpAuth());

    app.get('/mcp/tools/list', (req, res) => {
      const agent = (req as typeof req & { agent: AgentIdentity }).agent;
      res.json({ tools: [] });
    });

    app.listen(3000);
    ```
  </Tab>

  <Tab value="Next.js">
    `kavachNextjs` returns route handlers for the App Router. Use a catch-all route to handle all KavachOS paths.

    ```typescript
    // app/api/kavach/[...path]/route.ts
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachNextjs } from '@kavachos/nextjs';

    const kavach = createKavach({
      database: { provider: 'postgres', url: process.env.DATABASE_URL! },
      baseUrl: process.env.AUTH_BASE_URL!,
      mcp: {
        issuer: process.env.AUTH_BASE_URL!,
        audience: process.env.MCP_BASE_URL!,
      },
    });

    const mcp = createMcpModule(kavach);
    const handler = kavachNextjs(kavach, { mcp });

    export const GET = handler;
    export const POST = handler;
    export const DELETE = handler;
    ```

    To protect individual MCP routes, wrap the route handler:

    ```typescript
    // app/mcp/tools/route.ts
    import { kavachNextjs } from '@kavachos/nextjs';
    import type { NextRequest } from 'next/server';

    const { withMcpAuth } = kavachNextjs(kavach, { mcp });

    export const GET = withMcpAuth(async (req: NextRequest, { agent }) => {
      return Response.json({ tools: [] });
    });
    ```

    For middleware-level protection across all MCP routes:

    ```typescript
    // middleware.ts
    import { kavachNextjs } from '@kavachos/nextjs';
    import type { NextRequest } from 'next/server';

    const { mcpAuthMiddleware } = kavachNextjs(kavach, { mcp });

    export function middleware(req: NextRequest) {
      if (req.nextUrl.pathname.startsWith('/mcp')) {
        return mcpAuthMiddleware(req);
      }
    }
    ```
  </Tab>

  <Tab value="Fastify">
    `kavachFastify` returns a Fastify plugin. Register it with `fastify.register`.

    ```typescript
    import Fastify from 'fastify';
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachFastify } from '@kavachos/fastify';

    const kavach = createKavach({
      database: { provider: 'postgres', url: process.env.DATABASE_URL! },
      baseUrl: process.env.AUTH_BASE_URL!,
      mcp: {
        issuer: process.env.AUTH_BASE_URL!,
        audience: process.env.MCP_BASE_URL!,
      },
    });

    const mcp = createMcpModule(kavach);
    const fastify = Fastify();

    // Register the KavachOS plugin — mounts all routes under basePath
    await fastify.register(kavachFastify(kavach, { mcp }));

    // Protect MCP routes with a preHandler hook
    fastify.addHook('preHandler', kavachFastify(kavach, { mcp }).withMcpAuth());

    fastify.get('/mcp/tools/list', async (request) => {
      const agent = request.agent;   // AgentIdentity set by hook
      return { tools: [] };
    });

    await fastify.listen({ port: 3000 });
    ```
  </Tab>

  <Tab value="Nuxt">
    `kavachNuxt` returns a Nuxt server plugin. Add it to your server plugins directory.

    ```typescript
    // server/plugins/kavach.ts
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachNuxt } from '@kavachos/nuxt';

    const kavach = createKavach({
      database: { provider: 'postgres', url: process.env.DATABASE_URL! },
      baseUrl: process.env.AUTH_BASE_URL!,
      mcp: {
        issuer: process.env.AUTH_BASE_URL!,
        audience: process.env.MCP_BASE_URL!,
      },
    });

    const mcp = createMcpModule(kavach);

    export default kavachNuxt(kavach, { mcp });
    ```

    KavachOS routes are then available at `/api/kavach/...` within your Nuxt server.
  </Tab>

  <Tab value="SvelteKit">
    `kavachSvelteKit` returns a SvelteKit handle function. Compose it in `hooks.server.ts`.

    ```typescript
    // src/hooks.server.ts
    import { sequence } from '@sveltejs/kit/hooks';
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachSvelteKit } from '@kavachos/sveltekit';

    const kavach = createKavach({
      database: { provider: 'postgres', url: process.env.DATABASE_URL! },
      baseUrl: process.env.AUTH_BASE_URL!,
      mcp: {
        issuer: process.env.AUTH_BASE_URL!,
        audience: process.env.MCP_BASE_URL!,
      },
    });

    const mcp = createMcpModule(kavach);

    export const handle = sequence(kavachSvelteKit(kavach, { mcp }));
    ```

    Requests to `/api/kavach/...` are intercepted and handled by KavachOS before reaching your SvelteKit routes.
  </Tab>

  <Tab value="Astro">
    `kavachAstro` returns an Astro middleware function. Add it to your middleware chain.

    ```typescript
    // src/middleware.ts
    import { sequence } from 'astro:middleware';
    import { createKavach, createMcpModule } from 'kavachos';
    import { kavachAstro } from '@kavachos/astro';

    const kavach = createKavach({
      database: { provider: 'postgres', url: process.env.DATABASE_URL! },
      baseUrl: process.env.AUTH_BASE_URL!,
      mcp: {
        issuer: process.env.AUTH_BASE_URL!,
        audience: process.env.MCP_BASE_URL!,
      },
    });

    const mcp = createMcpModule(kavach);

    export const onRequest = sequence(kavachAstro(kavach, { mcp }));
    ```

## Using without an adapter

If your framework is not listed, use the core `McpModule` directly. It works with any server that handles standard `Request` objects.

```typescript
import { createKavach, createMcpModule } from 'kavachos';

const kavach = createKavach({ /* ... */ });
const mcp = createMcpModule(kavach);

async function handler(request: Request): Promise<Response> {
  // Let KavachOS handle its own routes
  const kavachResponse = await mcp.handleRequest(request);
  if (kavachResponse) return kavachResponse;

  // Validate auth for protected routes
  const authResult = await mcp.validateRequest(request);
  if (!authResult.valid) {
    return mcp.buildUnauthorizedResponse(request);
  }

  return new Response(JSON.stringify({ tools: [] }), {
    headers: { 'Content-Type': 'application/json' },
  });
}
```

<Callout type="info">
  `handleRequest` returns `null` when the request URL does not match any KavachOS route, so your own handlers run normally for all other paths.
</Callout>
